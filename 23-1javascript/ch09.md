# 구조분해 할당: ES6의 문법
## 객체 내부의 속성(키)와 할당받을 변수명이 같은 경우 생략 표기법
- const {body} = document;
  - const body = document.body;

- const obj = {a:1, b:2};
- const {a,b} = obj;  // 구조분해 할당
  - const a = obj.a;
  - const b = obj.b;
## 배열의 구조분해 할당
- const arr = [1,2,3];
- const {a,b,c} = arr;  // a = arr[0], b=arr[1],c=arr[2]
